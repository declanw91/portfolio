This is an example used for testing the buying functionality.
This file contains php for printing a 12x12 multiplication square. See below

<?php
	for($x = 0; $x <= 12; $x++) {
		for($y = 0; $y <= 12; $y++) {
 			$ans = ($x*$y);
			if($ans < 10) {
				echo "00".$ans. " |";
			}
			else if($ans < 100) {
				echo "0".$ans." |";
			}
			else {
				echo $ans. " |";
			}
		}
		echo '<br/>';
	}
?>
